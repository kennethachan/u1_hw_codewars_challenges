## Sum Arrays

let someNumbers = [1, 2, 3]
let sum = 0
for (let i = 0; i < someNumbers.length; i++) {
return sum += someNumbers[i]

}

## Reversed Strings

let world = ["world"]
function newWorld(world) {
return world.split("").reverse().join("")
}
